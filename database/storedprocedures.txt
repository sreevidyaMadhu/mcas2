CREATE DEFINER=`root`@`localhost` PROCEDURE `Insertproduct`(IN p_PtdID INT,IN p_Pname VARCHAR(100),IN p_Price DECIMAL(10,2),IN p_Quantity INT )
BEGIN
if p_Price > 0 and p_Quantity >=0 then
INSERT INTO Product(PtdID,Pname,Price,Quantity)
VALUES(p_PtdID,p_Pname,p_Price,p_Quantity);
END IF;

END
