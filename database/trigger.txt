use stores;
create table products(pdtid int primary key,pname varchar(50),price decimal(10,2),qtyinstock int);
create table sale(saleid int primary key,deliveryaddress varchar(100));
create table saleitem(saleid int,pdtid int,qty int,primary key(saleid,pdtid),foreign key (saleid) references sale(saleid),foreign key (pdtid) references products(pdtid));

DELIMITER //
create trigger UpdateAvailableQuantity 
after insert on saleitem
for each row
BEGIN
	update products
    set qtyinstock=qtyinstock-NEW.qty
    where pdtid=NEW.pdtid;
END //
DELIMITER ;
